if(NOT ${KK_EMULATOR})

  set(sources
      main.c
      signatures.c
      usb_flash.c)

  include_directories(
      ${CMAKE_SOURCE_DIR}/include
      ${CMAKE_BINARY_DIR}/include)

  set(linker_script ${CMAKE_CURRENT_SOURCE_DIR}/bootloader.ld)

  set(CMAKE_EXE_LINKER_FLAGS
      "${CMAKE_EXE_LINKER_FLAGS} -T${linker_script} -L${CMAKE_BINARY_DIR}/lib")

  add_executable(bootloader.elf ${sources})
  target_link_libraries(bootloader.elf
      kkboard
      kkboard.keepkey
      -Wl,--whole-archive kkvariant.keepkey -Wl,--no-whole-archive
      -Wl,--whole-archive kkvariant.salt -Wl,--no-whole-archive
      kktransport
      kkcrypto
      kkrand
      -lopencm3_stm32f2
      -lc
      -lm)
  add_custom_command(TARGET bootloader.elf
      POST_BUILD
      COMMAND ${CMAKE_OBJCOPY} ARGS -O binary
          ${CMAKE_BINARY_DIR}/bin/bootloader.elf
          ${CMAKE_BINARY_DIR}/bin/bootloader.bin)

endif()
